1. ฟังก์ชันอัปโหลดรูปภาพ + เพิ่มข้อมูลเสื้อผ้า (Insert)
import { supabase } from '../supabaseClient'; // Adjusted import path to match project structure

export const uploadAndSave = async (file, clothingData) => {
  try {
    // 1. อัปโหลดรูปภาพไปยัง Storage ก่อน
    const fileExt = file.name.split('.').pop();
    const fileName = `${Date.now()}.${fileExt}`; // ตั้งชื่อไฟล์ใหม่เพื่อหลีกเลี่ยงการซ้ำ
    const filePath = `clothing-images/${fileName}`;

    const { error: uploadError } = await supabase.storage
      .from('clothing-images')
      .upload(filePath, file);

    if (uploadError) throw uploadError;

    // 2. ดึง Public URL ของรูปที่อัปโหลดสำเร็จ
    const { data: urlData } = supabase.storage
      .from('clothing-images')
      .getPublicUrl(filePath);
    const imageUrl = urlData.publicUrl;

    // 3. บันทึกข้อมูลเสื้อผ้าลงตาราง clothes ของ Supabase
    const { error: insertError } = await supabase
      .from('clothes')
      .insert([
        {
          name: clothingData.name,
          pattern_type: clothingData.pattern_type, // 'solid' หรือ 'pattern'
          colors: clothingData.colors, // Array เช่น ['ขาว', 'ฟ้า']
          category: clothingData.category,
          owner: clothingData.owner,
          image_url: imageUrl,
          image_path: filePath
        }
      ]);

    if (insertError) throw insertError;
    alert('เพิ่มเสื้อผ้าเรียบร้อยแล้ว!');
  } catch (error) {
    console.error('Error:', error.message);
  }
};

2. ฟังก์ชันดึงข้อมูลมาแสดงผล + ค้นหา/ฟิลเตอร์ (Fetch & Filter)
import { ref } from 'vue';
import { supabase } from '../supabaseClient';

export const clothesList = ref([]);

export const fetchClothes = async (searchName = '', filterOwner = '', filterCategory = '', filterPattern = '', filterColors = []) => {
  let query = supabase.from('clothes').select('*');

  if (searchName) {
    query = query.ilike('name', `%${searchName}%`);
  }
  if (filterOwner) {
    query = query.eq('owner', filterOwner);
  }
  if (filterCategory) {
    query = query.eq('category', filterCategory);
  }
  if (filterPattern) {
    query = query.eq('pattern_type', filterPattern);
  }

  const { data, error } = await query.order('created_at', { ascending: false });
  if (!error) {
    // ถ้ามีการกรองสีเพิ่มเติม ให้ทำการ filter ที่ client side
    let result = data;
    if (filterColors.length) {
      if (filterPattern === 'solid') {
        // สีพื้น – ต้องตรงกับสีเดียว
        result = result.filter(item => item.colors.includes(filterColors[0]));
      } else if (filterPattern === 'pattern') {
        // ลาย – ต้องมีอย่างน้อย 1 สีที่ตรงกับ filterColors
        result = result.filter(item =>
          filterColors.some(col => item.colors.includes(col))
        );
      }
    }
    clothesList.value = result;
  }
};

3. ฟังก์ชันการลบข้อมูล (Delete)
export const deleteClothing = async (item) => {
  try {
    // 1. ลบรูปจาก Storage (ใช้ image_path ที่บันทึกไว้ใน DB)
    if (item.image_path) {
      await supabase.storage.from('clothing-images').remove([item.image_path]);
    }
    // 2. ลบแถวจากตาราง
    const { error } = await supabase
      .from('clothes')
      .delete()
      .eq('id', item.id);

    if (error) throw error;
    alert('ลบข้อมูลเรียบร้อยแล้ว');
  } catch (error) {
    alert('เกิดข้อผิดพลาดในการลบ');
  }
};
